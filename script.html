let pessoas = [] function adicionar() { let nome = document.getElementById("nome").value let idade = Number(document.getElementById("idade").value) if (nome === "" || idade <= 0) { alert("Preencha os campos corretamente") return } pessoas.push({ nome, idade }) let item = document.createElement("li") item.innerHTML = nome + " - " + idade + " anos" document.getElementById("lista").appendChild(item) document.getElementById("nome").value = "" document.getElementById("idade").value = "" } function calcularMedia() { if (pessoas.length === 0) { alert("Nenhuma pessoa cadastrada") return } let soma = 0 for (let i = 0; i < pessoas.length; i++) { soma += pessoas[i].idade } let media = soma / pessoas.length document.getElementById("resultado").innerHTML = "Média das idades: " + media.toFixed(1) + " anos" }
